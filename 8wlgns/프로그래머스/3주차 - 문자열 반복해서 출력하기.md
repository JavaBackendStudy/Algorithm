import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
        String str = sc.next();
        int n = sc.nextInt();
        
        StringBuffer result = new StringBuffer();
        
        for(int i=1; i<=n; i++) {
        	result.append(str);
        }
        
        System.out.print(result.toString());
    }
}
