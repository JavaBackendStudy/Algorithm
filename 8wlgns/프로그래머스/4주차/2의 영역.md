class Solution {
    public int[] solution(int[] arr) {
    	if(arr.length == 1 && arr[0] == 2) 
    		return new int[] {2};
    	
    	int left = -1;
    	int right = -1;
    	
    	for(int i=0; i<arr.length; i++)
    		if(arr[i] == 2) {
    			left = i;
    			break;
    		}
    	
    	for(int i=arr.length-1; i>=0; i--)
    		if(arr[i] == 2) {
    			right = i;
    			break;
    		}
    	
    	if(left == right) {
    		if(left == -1) {
    			return new int[] {-1};
    		} else {
    			return new int[] {2};
    		}
    	} else {
    		int[] answer = new int[right-left+1];
        	System.arraycopy(arr, left, answer, 0, right-left+1);	
        	return answer;
    	}
    }
}
