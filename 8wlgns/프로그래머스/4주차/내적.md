class Solution {
    public int solution(int[] a, int[] b) {
    	int total = 0;
    	
    	for(int i=0; i<a.length; i++) {
            if(a[i] == 0 || b[i] == 0) continue;
            
    		total += a[i]*b[i];
    	}
    	
    	return total;
    }
}
