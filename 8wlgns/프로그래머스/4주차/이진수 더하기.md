class Solution {
    public String solution(String bin1, String bin2) {
    	
    	int bin1Len = bin1.length();
    	int bin2Len = bin2.length();
    	
    	StringBuilder temp1 = new StringBuilder();
    	StringBuilder temp2 = new StringBuilder();
    	
    	if(bin1Len != bin2Len) {
    		if(bin1Len > bin2Len) {
    			int count = bin1Len - bin2Len;
    			while(count-- > 0) {
    				temp2.append('0');
    			}
    			for(char ch: bin2.toCharArray()) {
    				temp2.append(ch);
    			}
    			temp1.append(bin1);
    		} else {
    			int count = bin2Len - bin1Len;
    			while(count-- > 0) {
    				temp1.append('0');
    			}
    			for(char ch: bin1.toCharArray()) {
    				temp1.append(ch);
    			}
    			temp2.append(bin2);
    		}
    	} else {
    		temp1.append(bin1);
    		temp2.append(bin2);
    	}
    	
    	StringBuilder answer = new StringBuilder();
        String num = "";

        for(int i=temp1.length()-1; i>=0; i--) {
        	
        	if(temp1.charAt(i) == '0' && temp2.charAt(i) == '0') {
        		if("1".equals(num)) {
        			answer.append("1");
        			num = "";
        		} else {
        			answer.append("0");
        		}
        	} else if(temp1.charAt(i) == '1' && temp2.charAt(i) == '1') {
        		if("1".equals(num)) {
        			answer.append("1");
        			num = "1";
        		} else {
        			answer.append("0");
        			num = "1";
        		}
        	} else {
        		if("1".equals(num)) {
        			answer.append("0");
        			num = "1";
        		} else {
        			answer.append("1");
        		}
        	}
        }
        
        if("1".equals(num)) answer.append("1");
        
        return answer.reverse().toString();
    }
}
