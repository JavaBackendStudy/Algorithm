class Solution {
    public int solution(int[] sides) {
       int max = Math.max(sides[0], sides[1]);
       int min = Math.min(sides[0], sides[1]);
    	
       int result = 0;
       int num = 1;
       while(true) {
    	   if(num > max) break;
    	   if(num+min > max) result++;
    	   num++;
       }
       
       num=max+1;
       while(true) {
    	   if(max+min <= num) break;
    	   else result++;
    	   num++;
       }
       return result;
    }
}
