import java.util.LinkedList;
import java.util.Queue;

class Solution {
    boolean solution(String s) {
        Queue<Character> queue = new LinkedList();
        
        for(int i=0; i<s.length(); i++) {
     	   if (')' == s.charAt(i)) {
     		  if (queue.size() == 0) {
     			  return false;
     		  } else if('(' != queue.poll()) {
     			  return false;
     		  }
     	   } else {
     		  queue.add('('); 
     	   }
        }
        
        return queue.size() != 0 ? false : true;
    }
}
