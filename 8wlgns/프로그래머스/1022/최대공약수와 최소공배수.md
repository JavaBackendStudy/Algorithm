class Solution {
    public int[] solution(int n, int m) {
    	int gcd = 1;
        	
        for(int i=2; i<=Math.min(n, m); i++) {
        	while(n%i==0 && m%i==0) {
        		n/=i;
        		m/=i;
        		gcd *= i;
        	}
        }
            
        return new int[] {gcd, gcd*n*m};
    }
}
