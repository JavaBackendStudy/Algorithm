class Solution {
	public int solution(String myString, String pat) {
               char[] myChar = myString.toCharArray();
        char[] myPat = pat.toCharArray();
        
        int count = 0;
        int matchCount = 0;
        
        for(int i=0; i<=myChar.length - myPat.length; i++) {
        	matchCount = 0;
        	
        	int j=0;
        	for(char myEl: myPat) {
        		if(myEl== myChar[i+j++]) {
        			matchCount++;
        		}
        	}
        	
        	if(matchCount == myPat.length) {
        		count += 1;
        	}
        }
        
        return count;
    }
}
