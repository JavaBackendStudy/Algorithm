class Solution {
    public int solution(String my_string) {
        int answer = 0;
        
        StringBuilder sb = new StringBuilder(my_string);
        
        int calcMode = 0; // (0 -> none, 1 -> plus, 2 -> minus)
        int offSet = 0;
        
        for(int i=0; i<sb.length(); i=offSet){
            if(sb.charAt(i) == '+'){
                calcMode = 1;
                offSet++;
                continue;
            }
            else if(sb.charAt(i) == '-'){
                calcMode = 2;
                offSet++;
                continue;
            }
            else if(sb.charAt(i) == ' '){
                offSet++;
                continue;
            }
            
            int digitCnt = offSet;
            for(int j=offSet; j<sb.length(); j++){
                if(sb.charAt(j) >= '0' && sb.charAt(j) <= '9') digitCnt++;
                else break;
            }
            
            switch(calcMode){
                case 0:
                    answer += Integer.parseInt(sb.substring(offSet, digitCnt));
                    offSet = digitCnt;
                    break;
                case 1:
                    answer += Integer.parseInt(sb.substring(offSet, digitCnt));
                    offSet = digitCnt;
                    break;
                case 2:
                    answer -= Integer.parseInt(sb.substring(offSet, digitCnt));
                    offSet = digitCnt;
                    break;
            }
        }
        
        return answer;
    }
}
