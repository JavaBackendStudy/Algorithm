import java.util.stream.IntStream;

class Solution {
    public int[] solution(String my_string) {
       final int MAX = 52;
       int[] answer = new int[MAX];
       
       int[] list = IntStream.range(0, my_string.length())
       .map(i -> my_string.charAt(i) - 'A')
       .toArray();
       
       for(int idx: list) {
    	   if(idx >= 0 && idx <= 25)
    		   answer[idx]++;
    	   else
    		   answer[idx-6]++;
       }
    	   
       return answer;
    }
}
