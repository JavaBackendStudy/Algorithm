    public int solution(int[] numbers, int k) {
    	int curr_index = 0;
    	int len = numbers.length;
    	
    	while(k-- > 1) {
    		if(curr_index+2 < len) {
    			curr_index += 2;
    		}else {
    			curr_index -= (len-1);

    		}
    	}
