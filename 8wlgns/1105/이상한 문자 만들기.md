package leetCode;

import java.util.Arrays;

class Solution {
	
	public static void main(String[] args) {
		
		int n = 15;
		
		long beforeTime = System.currentTimeMillis(); //코드 실행 전에 시간 받아오기
		String answer = solution("HELLO");
		
		long afterTime = System.currentTimeMillis(); // 코드 실행 후에 시간 받아오기
		long diffTime = afterTime - beforeTime; // 두 개의 실행 시간
		System.out.println("실행 시간(ms): " + diffTime); // 세컨드(초 단위 변환)
		System.out.println(answer);
	}
	
	/*
	 *  O(n)
	 *  start라는 변수를 0으로 셋팅한다.
	 *  s를 문자가 담긴 배열로 변환후 이를 순회한다.
	 *  공백이라면 start변수를 0으로 업데이트한다.
	 *  아니라면 현재의 위치가 2로 나눈 나머지가 0이고, 소문자라면 대문자로 변환한다.
	 *  그렇지 않다면 (홀수라면) 대문자라면 소문자로 변환한다.
	 */
    public static String solution(String s) {
    	StringBuilder sb = new StringBuilder();
        char[] arr = s.toCharArray();
        
        int start = 0;
        for (int i=0; i<arr.length; i++) {
        	if (arr[i] == ' ') start = 0;
        	else {
        		if (start%2 == 0 && 
        				arr[i] >= 'a' && arr[i] <='z') {
        			arr[i] = (char) ('A' + (arr[i] - 'a'));
        			
        		} else if (start%2 != 0 && 
        				arr[i] >= 'A' && arr[i] <= 'Z') {
        			arr[i] = (char) ('a' + (arr[i] - 'A'));
        		}
        		start++;
        	}
        	
        	sb.append(arr[i]);
        }
        
        return sb.toString();
    }
}
