class Solution {
	public int calcGcd(int a, int b) {
		int min = Math.min(a, b);
		int answer = 0;
		
		for(int i=2; i<=min; i++) {
			if(a%i == 0 && b%i == 0) {
				answer = i;
			}
		}
		
		return answer;
	}
	
	public int solution(int a, int b) {
		int gcd = calcGcd(a, b);
		if(gcd > 0) {
			a /= gcd;
			b /= gcd;
		}
		
		int[] num = {2, 5};
		for(int n: num) {
			while(b >= n && b%n == 0) {
				b /= n;
			}
		}
		
		return b == 1 ? 1 : 2;
	}
}
