class Solution {
	public static int countOneNumber(int n) {
		int len = 0;
		while(n>0) {
			if(n%2 == 1) len++;
			n/=2;
		}
		return len;
	}
	
    public static int solution(int n) {
    	int answer = 0;
    	int len = countOneNumber(n);
    	
    	for(int i=n+1; i<=n*2; i++) {
    		if (len == countOneNumber(i)) {
    			answer = i;
    			break;
    		}
    	}
    	
    	return answer;
    }
}
