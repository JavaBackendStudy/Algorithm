##### **📜 대략적인 코드 설명**
```java
package leetCode;

class Solution {
	
	public static void main(String[] args) {
		
		int chicken = 15; 
		int answer = solution(chicken);
		
		System.out.println(answer);
		
	}
	
	/*
	 * n부터 최대 n*2까지 순회하는 for loop를 선언한다.
	 *  1) n을 2진수에 대한 1의 갯수를 센다. (logN)
	 *  2) 1의 갯수가 n의 1의 갯수와 같다면 loop를 나간다.
	 *  
	 * 시간복잡도: O(nlogN) -> 최대 1000000 * 19의 연산을 수행
	 */
	
	public static int calc(int n) {
		int len = 0;
		while(n>0) {
			if(n%2 == 1) len++;
			n/=2;
		}
		return len;
	}
	
    public static int solution(int n) {
    	int answer = 0;
    	int len = calc(n);
    	
    	for(int i=n+1; i<=n*2; i++) {
    		if(len == calc(i)) {
    			answer = i;
    			break;
    		}
    	}
    	
    	return answer;
    }
}
```
