	public static int calc(int n) {
		int len = 0;
		while(n>0) {

		}

    public static int solution(int n) {
    	int answer = 0;
    	int len = calc(n);
    	
    	for(int i=n+1; i<=n*2; i++) {
    		if(len == calc(i)) {
    			answer = i;
    			break;
    		}
    	}
    	
    	return answer;
