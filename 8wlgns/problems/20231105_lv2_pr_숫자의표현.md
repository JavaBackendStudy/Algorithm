  자릿수는 크게 중요하지 않았고, 1부터 다음 값을 누적해 n과 같다면 answer++하고,
  그 다음값에서 다시 시작해 누적하는 식으로 문제를 해결하면 됬었다.

class Solution {
    public int solution(int n) {
        int answer = 1; // n 자릿수 한개만 더하는 경우
        
        int count = 0;
        int idx = 0;
        for(int i=2; i<=n/2+1; i++) {
        	
        	for(int j=1; j<=n/2+1; j++) {
        		
        		count = 0;
        		for(int k=j; k<j+i; k++) {
        			if(count > n) break;
        			count += k;
        		}
        		if(n == count) {
                    answer++;
                    break;
                }
        	}
            
        }
        
        return answer;
    }
}

class Solution {
    public int solution(int n) {
        int answer = n/2+1 < n ? 1 : 0; 
        int count = 0;
        int number = 0;
        
        for(int i=1; i<=n/2+1; i++) {
        	
        	count = 0;
        	number = i;
        	
        	while(true) {
        		if(count >= n) {
        			if (count == n) answer++;
        			break;
        		}
        		else {
        			count += number++;
        		}
        	}
        }
        
        return answer;  

    }
}
