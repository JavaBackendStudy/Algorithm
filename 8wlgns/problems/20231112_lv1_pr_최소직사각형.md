class Solution {
    public int solution(int[][] sizes) {
        int maxW = 0;
        int maxH = 0;
        
        int currW, currH, temp;
        for(int i=0; i<sizes.length; i++){
            currW = sizes[i][0];
            currH = sizes[i][1];
            
            if(currW < currH){
                temp = currH;
                currH = currW;
                currW = temp;
            }
            
            if(maxW < currW) maxW = currW;
            if(maxH < currH) maxH = currH;
        }
        
        return maxW * maxH;
    }
}
