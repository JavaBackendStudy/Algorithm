class Solution {
    public String solution(String s) {
    	StringBuilder sb = new StringBuilder();
        char[] arr = s.toCharArray();
        
        int wordIdx = 0;
        for (int i=0; i<arr.length; i++) {
        	if (arr[i] == ' ') {
                wordIdx = -1;
            } else if (wordIdx%2 == 0 && 
        				arr[i] >= 'a' && 
                         		arr[i] <='z') {
        			arr[i] = (char) ('A' + (arr[i] - 'a'));
            } else if (wordIdx%2 != 0 && 
        				arr[i] >= 'A' && 
                         		arr[i] <= 'Z') {
        			arr[i] = (char) ('a' + (arr[i] - 'A'));
        	}
            sb.append(arr[i]);
        	wordIdx++;
        }
        
        return sb.toString();
    }
}
