class Solution {
    public int solution(int n) {
    	int answer = 0;
        int count = 0;
        
        for(int i=1; i<=n*2; i++) {
        	String num = String.valueOf(i);
        	if(num.contains("3") || i%3 == 0) continue;
        	count++;
        	
        	if(count == n) {
        		answer = i;
        		break;
        	}
        }
        
        return answer;
    }
}
